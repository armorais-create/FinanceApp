    });
    window.addEventListener("unhandledrejection", function (e) {
      document.getElementById("log").innerHTML += "Promise Error: " + e.reason + "<br>";
      fetch("/log_error?msg=" + encodeURIComponent(e.reason));
    });
  </script>
